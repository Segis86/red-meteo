const express = require('express'); const app = express(); const PORT = process.env.PORT || 3000; // Para poder recibir JSON desde Ecowitt app.use(express.json()); app.use(express.urlencoded({ extended: true })); // Inicializamos la lista de estaciones let estaciones = []; // Endpoint que Ecowitt usará para enviar datos app.post('/update', (req, res) => { const body = req.body; // body puede tener diferentes nombres según tu configuración Ecowitt // Ejemplo esperado: // { // "station_id": "IVILLA1255", // "nombre": "Villar del Arzobispo - Iglesia", // "lat": 39.73, // "lon": -0.82, // "temp": 15, // "hum": 65, // "wind_speed": 10, // "wind_dir": 135 // } if(!body.station_id) return res.status(400).send("Falta station_id"); // Buscar si ya existe la estación let index = estaciones.findIndex(e => e.station_id === body.station_id); const estData = { station_id: body.station_id, nombre: body.nombre || "Estación "+body.station_id, lat: parseFloat(body.lat) || 0, lon: parseFloat(body.lon) || 0, link: `https://www.wunderground.com/dashboard/pws/${body.station_id}`, temp: parseFloat(body.temp) || 0, hum: parseFloat(body.hum) || 0, wind_speed: parseFloat(body.wind_speed) || 0, wind_dir: parseFloat(body.wind_dir) || 0 }; if(index >= 0){ // Actualizar estación existente estaciones[index] = estData; } else { // Añadir nueva estación estaciones.push(estData); } res.send({status: "ok", estaciones}); }); // Endpoint público para tu web app.get('/estaciones.json', (req, res) => { res.json(estaciones); }); // Servir puerto app.listen(PORT, () => { console.log(`Servidor activo en puerto ${PORT}`); });
